# TOP 200 SCRAPE

url = "https://en.wikipedia.org/wiki/List_of_top_1000_universities_in_the_world"

r = requests.get(url)
soup = BeautifulSoup(r.content, "html.parser")

# UNI NAME

tr_group = []
td_group = []
# result = soup.find_all('tr', table="wikitable sortable jquery-tablesorter")
for tr_tag in soup.findAll('tr', table="wikitable sortable jquery-tablesorter"):
    tr_group.append(tr_tag)

for td_tag in soup.findAll('td', class_="mw-parser-output"):
    td_group.append(td_tag)

for n in tr_group:
    print(tr_group[n])
